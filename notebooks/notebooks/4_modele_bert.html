# Tokenisation manuelle en batch pour viter les problmes de mmoire
def tokenize_data(texts, labels, tokenizer, max_length):
    # Convertir tous les textes en string
    texts = [str(text) for text in texts]
    
    # Tokeniser par lot
    encodings = tokenizer(
        texts,
        truncation=True,
        padding='max_length',
        max_length=max_length,
        return_tensors='tf'
    )
    
    # Crer un dataset TensorFlow
    dataset = tf.data.Dataset.from_tensor_slices((
        {
            'input_ids': encodings['input_ids'],
            'attention_mask': encodings['attention_mask'],
            'token_type_ids': encodings['token_type_ids']
        },
        labels
    ))
    
    return dataset

# Tokeniser les donnes

val_dataset = tokenize_data(X_val_sample, y_val_sample, tokenizer, MAX_SEQ_LENGTH)
test_dataset = tokenize_data(X_test_sample, y_test_sample, tokenizer, MAX_SEQ_LENGTH)

# Configuration des datasets pour l'entranement

# Chargement du modle BERT pr-entran
print("Chargement du modle BERT pr-entran...")
bert_model = TFBertForSequenceClassification.from_pretrained(
    'bert-base-uncased', 
    num_labels=2,
    cache_dir='models/bert/bert_cache'
)

# Configuration des optimiseurs avec un learning rate plus faible
optimizer = tf.keras.optimizers.Adam(learning_rate=1e-5, epsilon=1e-08)

# Compilation du modle
bert_model.compile(
    optimizer=optimizer,
    loss=tf.keras.losses.SparseCategoricalCrossentropy(from_logits=True),
    metrics=['accuracy']
)

# Dfinition des callbacks avec plus de patience pour conomiser des ressources
callbacks = [
    tf.keras.callbacks.EarlyStopping(
        monitor='val_loss', 
        patience=2,  # Rduit pour arrter plus tt si pas d'amlioration
        restore_best_weights=True
    ),
    # Rduire le nombre de callbacks pour simplifier
    tf.keras.callbacks.ReduceLROnPlateau(
        monitor='val_loss', 
        factor=0.5, 
        patience=1, 
        min_lr=1e-6
    )
]

# Fine-tuning du modle BERT avec MLflow tracking
print("Dbut du fine-tuning du modle BERT...")
EPOCHS = 10  

with mlflow.start_run(run_name="BERT_base_fine_tuning"):
    # Enregistrement des paramtres
    mlflow.log_param("model_type", "BERT-base-uncased")
    mlflow.log_param("batch_size", BATCH_SIZE)
    mlflow.log_param("max_sequence_length", MAX_SEQ_LENGTH)
    mlflow.log_param("learning_rate", 1e-5)
    mlflow.log_param("sample_size", sample_size)
    
    # Entranement
    start_time = time.time()
    history = bert_model.fit(
        train_dataset,
        validation_data=val_dataset,
        epochs=EPOCHS,
        callbacks=callbacks,
        verbose=1
    )
    training_time = time.time() - start_time
    
    # Librer la mmoire aprs l'entranement
    gc.collect()
    
    # valuation sur les donnes de validation
    val_loss, val_accuracy = bert_model.evaluate(val_dataset, verbose=1)
    
    # Prdictions (version simplifie)
    y_pred_logits = []
    y_val_true_list = []
    
    for batch in val_dataset:
        inputs, labels = batch
        outputs = bert_model(inputs)
        logits = outputs.logits
        predictions = tf.argmax(logits, axis=-1)
        
        y_pred_logits.extend(predictions.numpy())
        y_val_true_list.extend(labels.numpy())
    
    y_pred = np.array(y_pred_logits)
    y_val_true = np.array(y_val_true_list)
    
    # Mtriques de performance
    val_precision = precision_score(y_val_true, y_pred, average='binary')
    val_recall = recall_score(y_val_true, y_pred, average='binary')
    val_f1 = f1_score(y_val_true, y_pred, average='binary')
    
    # Enregistrement des mtriques
    mlflow.log_metric("val_loss", val_loss)
    mlflow.log_metric("val_accuracy", val_accuracy)
    mlflow.log_metric("precision", val_precision)
    mlflow.log_metric("recall", val_recall)
    mlflow.log_metric("f1", val_f1)
    mlflow.log_metric("training_time", training_time)
    
    # Sauvegarde du modle
    mlflow.keras.log_model(bert_model, "BERT_fine_tuned")
    
    # Visualisation de l'apprentissage

    plt.xlabel('Epoch')
    plt.ylabel('Loss')
    plt.legend(['Train', 'Validation'], loc='upper right')
    plt.tight_layout()
    plt.show()
    
    # Affichage des rsultats
    print("\nRsultats du fine-tuning BERT:")

    print(f"Precision: {val_precision:.4f}")
    print(f"Recall: {val_recall:.4f}")
    print(f"F1-score: {val_f1:.4f}")
    print(f"Temps d'entranement: {training_time:.2f} secondes")
    
    # Matrice de confusion
    cm = confusion_matrix(y_val_true, y_pred)
    plt.figure(figsize=(8, 6))
    sns.heatmap(cm, annot=True, fmt='d', cmap='Blues',

                yticklabels=['Ngatif', 'Positif'])
    plt.xlabel('Prdiction')
    plt.ylabel('Rel')
    plt.title('Matrice de confusion - BERT fine-tun')

    plt.show()

# Fonction pour convertir les exemples en features compatibles avec DistilBERT
def convert_examples_to_tf_dataset_distilbert(texts, labels, tokenizer, max_length=MAX_SEQ_LENGTH):
    """
    Convertit une liste de textes et d'tiquettes en un dataset TensorFlow compatible avec DistilBERT.
    Version optimise pour utiliser moins de mmoire.
    """
    # Convertir tous les textes en string
    texts = [str(text) for text in texts]
    
    # Tokeniser par lot pour conomiser du temps
    encodings = tokenizer(
        texts,
        add_special_tokens=True,
        max_length=max_length,
        padding='max_length',
        truncation=True,
        return_attention_mask=True,
        return_tensors='tf'
    )
    
    # Crer un dataset TensorFlow directement  partir des tenseurs
    dataset = tf.data.Dataset.from_tensor_slices((
        {
            'input_ids': encodings['input_ids'],
            'attention_mask': encodings['attention_mask']
        },
        labels
    ))
    
    return dataset

            self.d_model = d_model
            self.d_model = tf.cast(self.d_model, tf.float32)
            self.warmup_steps = warmup_steps
            
        def __call__(self, step):
            step = tf.cast(step, tf.float32)
            arg1 = tf.math.rsqrt(step)
            arg2 = step * (self.warmup_steps ** -1.5)
            
            return tf.math.rsqrt(self.d_model) * tf.math.minimum(arg1, arg2)

    learning_rate = CustomSchedule(768)  # 768 est la dimension du modle DistilBERT
    optimizer = tf.keras.optimizers.Adam(learning_rate, beta_1=0.9, beta_2=0.98, epsilon=1e-9)

# Compilation du modle DistilBERT
distilbert_model.compile(
    optimizer=optimizer,
    loss=tf.keras.losses.SparseCategoricalCrossentropy(from_logits=True),

# Cration d'une fonction de prdiction qui sera utilise par l'API
def predict_sentiment_bert_api(tweet, model=best_bert_model, tokenizer=best_bert_tokenizer, max_length=MAX_SEQ_LENGTH):
    """
    Fonction pour prdire le sentiment d'un tweet avec BERT, utilisable par l'API.
    
    Args:
        tweet (str): Le tweet  analyser
        model: Le modle BERT ou DistilBERT charg
        tokenizer: Le tokenizer correspondant
        max_length (int): La longueur maximale des squences
        
    Returns:
        dict: Dictionnaire contenant la prdiction et les informations associes
    """
    # Encodage du tweet
    encoding = tokenizer.encode_plus(
        tweet,
        add_special_tokens=True,
        max_length=max_length,
        padding='max_length',
        truncation=True,
        return_attention_mask=True,
        return_token_type_ids='token_type_ids' in tokenizer.model_input_names,
        return_tensors='tf'
    )
    
    # Prparation des entres pour le modle
    inputs = {
        'input_ids': encoding['input_ids'],
        'attention_mask': encoding['attention_mask']
    }
    
    if 'token_type_ids' in encoding:
        inputs['token_type_ids'] = encoding['token_type_ids']
    
    # Prdiction
    prediction = model.predict(inputs)
    logits = prediction[0]
    probabilities = tf.nn.softmax(logits, axis=1).numpy()[0]
    predicted_class = np.argmax(probabilities)
    
    # Cration du rsultat
    result = {
        'tweet': tweet,
        'sentiment': 'Positif' if predicted_class == 1 else 'Ngatif',
        'confidence': float(probabilities[predicted_class])
    }
    
    return result
